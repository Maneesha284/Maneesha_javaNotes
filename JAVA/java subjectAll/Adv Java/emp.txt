public class RegistrationEmp extends GenericServlet {
	Connection con=null;
	Statement st=null;
	ResultSet rs=null;
	@Override
	public void service(ServletRequest req,ServletResponse res) throws ServletException, IOException {
		try {
			Class.forName("oracle.jdbc.driver.OracleDriver");
			con=DriverManager.getConnection("jdbc:oracle:thin:off/off@localhost:1521:xe");
			st=con.createStatement();
			String userid=req.getParameter("userid");
			String addressid=req.getParameter("addressid");
			String addresstype=req.getParameter("addresstype");
			String houseno=req.getParameter("houseno");
			String street=req.getParameter("street");
			String city=req.getParameter("city");
			String state=req.getParameter("state");
			String country=req.getParameter("country");
			String pincode=req.getParameter("pincode");
			String phno=req.getParameter("phno");
			String insertQuery = "insert into addresses values(?,?,?,?,?,?,?,?,?,?)";
			PreparedStatement ps=con.prepareStatement(insertQuery); 
			ps.setString(1,userid);
			ps.setString(2,addressid);
			ps.setString(3,addresstype);
			ps.setString(1,houseno);
			ps.setString(2,street);
			ps.setString(3,city);
			ps.setString(1,state);
			ps.setString(2,country);
			ps.setString(3,pincode);
			ps.setString(1,phno);
			ps.execute();
			PrintWriter out=res.getWriter();
			res.setContentType("text/html");
			out.println("<html>");
			out.println("<body bgcolor='cyan'>");
			out.println("<h1>hello "+userid+" !you are registered successfully. </h1>");
			out.println("</body>");
			out.println("</html>");
		}
		catch(Exception e) {
			System.out.println("===========");
		}
	}
}
